```TS
"use server"

import { readFile, writeFile } from "./helpers";

export async function createTicket(data:FormData){
  const name=data.get("name") as string;
  const type=data.get("type") as string;
  const status=data.get("status") as string;
  
  const tickets=readFile();
  const id=tickets.length+1;

  const ticket={ id, name, type, status }
  writeFile(ticket);
}
```

###### Server actions are mainly, for now, used to perform mutation from client side as they make a POST(form etc) request
###### If you want to call server action from the client side, you'll have to create a separate file with "use server" written at the start of the file.
```TS
// lib/action.ts
"use server"
export async function mutate(){}
```
###### To call server action from the server side, you can just write "use server" at top of the function body.

```TS
async function handleSubmit(){
	"use server"
	...
}
```
