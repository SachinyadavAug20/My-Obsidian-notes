###### This error happens when we try to render an image from unknown source in our Image component from NextJS
###### This error can be solved by telling nextJS to allow that image(as it does optimization and it cost power so need to give permission to it)
###### ![[Pasted image 20260426143817.png]] go to next config. Add the image URL
```TS
import type { NextConfig } from "next";

const nextConfig: NextConfig = {
  /* config options here */
  images:{
    remotePatterns:[
      {
        protocol: 'https',
        hostname: 'avatars.githubusercontent.com',
      }
    ]
  }
};

export default nextConfig;
```
