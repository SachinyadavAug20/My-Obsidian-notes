### Get by ID
```TS
import tickets from "@/Database/tickets";
import { NextResponse } from "next/server";

// app/api/tickets/[id]/route.ts
export async function GET( request: Request, { params }: { params: Promise<{ id: string }> },) {
  const { id } = await params;
  const response = tickets.filter((t) => t.id === parseInt(id));
  return NextResponse.json(response);
}
```

## Update
###### go to same URL and do a update request 
```TS
export async function PUT( request: Request, { params }: { params: Promise<{ id: string }> },) {
  const {id}=await params;
  const {name,status,type}=await request.json();
  const ticket:{id:number,name:string,status:string,type:string}|undefined=tickets.find((t)=>t.id===parseInt(id));
  if(!ticket) return NextResponse.json({message:"Ticket not found"},{status:404});

  if(name) ticket.name=name;
  if(status) ticket.status=status;
  if(type) ticket.type=type;
  return NextResponse.json(ticket);
}
```

## Delete request
```TS
export async function DELETE( _: Request, { params }: { params: Promise<{ id: string }> },) {
  const { id } = await params;
  const ticketIndex = tickets.findIndex((t) => t.id === parseInt(id));
  if (!ticketIndex)
    return NextResponse.json({ message: "Ticket not found" }, { status: 404 });
  tickets.splice(ticketIndex, 1);
  return NextResponse.json(tickets);
}
```

## Search
###### Search params --> it is by using URL params.![[Pasted image 20260429150307.png]] 
```TS
import tickets from "@/Database/tickets";
import { NextRequest, NextResponse } from "next/server";

// path : api/tickets/search?query=hello

export async function GET(request:NextRequest){
  const searchParams=request.nextUrl.searchParams;
  const query=searchParams.get("query");

  if(!query) return NextResponse.json({message:"No query found"},{status:400});
  const filteredquery=tickets.filter((ticket)=>ticket.name.toLowerCase().includes(query.toLowerCase()))
  return NextResponse.json(filteredquery)
}
```
